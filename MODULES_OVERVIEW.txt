🚀 USAGE EXAMPLES
================================================================================

// Access the API
const AUROMA = window.AUROMA;

// Drawing
AUROMA.updateBrushSize(300);
AUROMA.setBrushShape('circle');
AUROMA.rotateBrush(Math.PI / 4);

// Effects
AUROMA.toggleEffect('neon', true);
AUROMA.toggleEffect('chromatic', true);

// History
AUROMA.undo();
AUROMA.redo();

// Blockchain
await AUROMA.connectWallet();
await AUROMA.mintERC721(metadata, imageData);

// Canvas
AUROMA.loadDroppedImage(file, 'base');
AUROMA.clearCanvas('paint');

// State
console.log(AUROMA.brushState.brushSize);
console.log(AUROMA.effectStates.isNeonHeld);
